import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class FactFrame extends JFrame implements ActionListener {
    JTextField t1, t2;
    JButton b0;

    public FactFrame() {
        setTitle("Factorial Calculator");
        setSize(300, 200);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new GridLayout(3, 2));

        add(new JLabel("Enter an Integer:"));
        t1 = new JTextField(10);
        add(t1);

        add(new JLabel("Factorial Value:"));
        t2 = new JTextField(10);
        t2.setEditable(false);
        add(t2);

        b0 = new JButton("Compute");
        b0.addActionListener(this);
        add(b0);
    }

    public void actionPerformed(ActionEvent e) {
        int n = Integer.parseInt(t1.getText());
        int f = 1;
        for (int i = 1; i <= n; i++) {
            f *= i;
        }
        t2.setText(String.valueOf(f));
    }

    public static void main(String[] args) {
        FactFrame frame = new FactFrame();
        frame.setVisible(true);
    }
}

Installing Java and Eclipse

Java Installation

  1. Download the latest JDK from the Oracle website or use OpenJDK.
  2. Install the JDK by following the installation wizard.
  3. Set up the environment variables:
    • Windows: Add JAVA_HOME and update the PATH variable.
    • Linux/macOS: Add export JAVA_HOME="/path/to/jdk" in ~/.bashrc or ~/.zshrc.

Eclipse Installation

  1. Download Eclipse IDE from the official website.
  2. Extract the downloaded file and run the Eclipse installer.
  3. Choose the "Eclipse IDE for Java Developers" option and complete the installation.
  4. Open Eclipse and set up a new Java project.
