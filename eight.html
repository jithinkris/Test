// Step 1: Create an interface
interface Service {
    void request();
}

// Step 2: Create a RealService class (Actual implementation)
class RealService implements Service {
    public void request() {
        System.out.println("Real Service: Handling request...");
    }
}

// Step 3: Create a ProxyService class (Proxy to RealService)
class ProxyService implements Service {
    private RealService realService;

    public void request() {
        if (realService == null) {
            realService = new RealService(); // Create object only when needed
        }
        System.out.println("Proxy: Forwarding request to Real Service");
        realService.request();
    }
}

// Step 4: Client Code
public class ProxyPatternSimple {
    public static void main(String[] args) {
        Service proxy = new ProxyService();
        proxy.request(); // First time, real service is created
        System.out.println();
        proxy.request(); // Second time, real service is reused
    }
}

Installing Java and Eclipse

Java Installation

  1. Download the latest JDK from the Oracle website or use OpenJDK.
  2. Install the JDK by following the installation wizard.
  3. Set up the environment variables:
    • Windows: Add JAVA_HOME and update the PATH variable.
    • Linux/macOS: Add export JAVA_HOME="/path/to/jdk" in ~/.bashrc or ~/.zshrc.

Eclipse Installation

  1. Download Eclipse IDE from the official website.
  2. Extract the downloaded file and run the Eclipse installer.
  3. Choose the "Eclipse IDE for Java Developers" option and complete the installation.
  4. Open Eclipse and set up a new Java project.
